def create_images(column_name, img):
  output_images=[]
  print("column name" , column_name)
  for row_index,row in img.iterrows():
            outputImage = np.zeros((128, 128, 3), dtype="uint8")
            image_temp1 = cv2.imread(getattr(row,column_name))
            image1 = cv2.resize(image_temp1, (128 , 128))       
            outputImage[0:128, 0:128] = image1
            output_images.append(outputImage)  
  return output_images
